let contentInterval;let isTouching = false; // Статус, удерживается ли палец на экране // Функция для автоматической генерации контента function generateAIContent() { const randomContent = contentExamples[Math.floor(Math.random() * contentExamples.length)]; const aiContentContainer = document.getElementById('ai-content'); const newContent = document.createElement('div'); newContent.innerHTML = `

${randomContent.title}

${randomContent.content}

`; aiContentContainer.innerHTML = ''; // Очищаем предыдущий контент aiContentContainer.appendChild(newContent); } // Функция для старта обновления контента function startContentUpdate() { // Обновление контента каждые 15 секунд contentInterval = setInterval(generateAIContent, 15000); } // Функция для остановки обновления function stopContentUpdate() { clearInterval(contentInterval); } // Слушатели событий для остановки и возобновления обновления при касании экрана document.body.addEventListener('touchstart', () => { isTouching = true; // Если экран удерживается stopContentUpdate(); // Останавливаем обновление }); document.body.addEventListener('touchend', () => { isTouching = false; // Если палец убран if (!isTouching) { startContentUpdate(); // Возобновляем обновление } }); // Инициализация контента и запуск таймера generateAIContent(); startContentUpdate();
